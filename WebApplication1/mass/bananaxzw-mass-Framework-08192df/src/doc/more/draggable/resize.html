通过拖动右下角调整盒子大小。 返回

            $.require("ready,more/draggable",function(){
                $(".drag").draggable({
                    dragstart:function( e, dd ){
                        dd.width = $( this ).width();
                        dd.height = $( this ).height();
                    },
                    lockX: true,
                    lockY: true,
                    handle:'.handle',
                    drag: function(e, dd){
                        $( this ).css({
                            width: Math.max( 20, dd.width + dd.deltaX ),
                            height: Math.max( 20, dd.height + dd.deltaY )
                        });
                    }
                });
            });
