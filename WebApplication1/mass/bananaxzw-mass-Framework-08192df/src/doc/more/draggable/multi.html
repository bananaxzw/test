通过点击选中一起要拖动的方块，再拖动它们。 返回

            $.require("ready,more/draggable",function(){
                $(".drag").click(function(){
                    $( this ).toggleClass("selected");
                }).draggable({
                    dragstart:function(e, dd){
                        dd.multi = $('.selected');
                    }
                });
            });
