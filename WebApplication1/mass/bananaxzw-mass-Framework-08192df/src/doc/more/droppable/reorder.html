                $("li").droppable({
                    drop:function(event,dd){
                        var fn = dd.dropper.data("_insert");
                        if(fn ){
                            dd.dragger[fn](dd.dropper);
                        }
                    },
                    range:"li",
                    tolerance: function( event, dragger, dropper ){
                        var test = event.pageY > ( dropper.top + dropper.height / 2 );
                        dropper.elem.data("_insert", test ? "afterTo" : "beforeTo")
                        return this.contains( dropper, [ event.pageX, event.pageY ] );
                    }
                });
