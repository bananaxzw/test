 18 SL().create(function (SL) {
 19 
 20     /**
 21     * Event
 22     * @class
 23     *@name Event
 24     */
 25     var Event = function () {
 26         //事件注册函数的唯一标识
 27         this.guid = 1;
 28     };
 29     Event.prototype = {
 30         /**
 31         *注册事件
 32         *@param {domElement} element dom元素
 33         *@param {String} type 事件类型 
 34         *@param {Function} handler 事件处理函数
 35         *@param {Object} data 事件扩展数据
 36         *@example 
 37         * var tt = document.getElementById("sff");
 38         * SL().Event.addEvent(tt, "click", function (event) {
 39         *    console.log(event.extendData.name);
 40         *}, { "name": "xuzhiwei" });
 41         *SL().Event.addEvent(tt, "click", function (event) {
 42         *    console.log(event.extendData.name);
 43         *}, { "name": "xuzhiwei1" });
 44         *function see() {
 45         *    alert("i see you!");
 46         *}
 47         *SL().Event.addEvent(tt, "click", see);
 48         */
 49         addEvent: function (element, type, handler, data) {
 50             var _this = this;
 51             //给函数分配唯一的标志ID
 52             if (!handler.$$guid) handler.$$guid = this.guid++;
 53             //创建一个hash table来保存各种事件的处理函数  
 54             element.events = element.events || {};
 55             //创建一个hash table来保存某个事件处理函数
 56             var handlers = element.events[type];
 57             if (!handlers) {
 58                 handlers = element.events[type] = {};
 59                 //储存已经存在的事件处理函数
 60                 if (element["on" + type]) {
 61                     handlers[0] = { "handler": element["on" + type], "data": null };
 62                 }
 63             }
 64             // 保存时间处理函数到hash table中
 65             handlers[handler.$$guid] = { "handler": handler, "data": data };
 66             // 为事件提供一个统一全局的处理函数 这句是关键handleEvent函数属于element，也就是说函数的内部this是指向element
 67             element["on" + type] = handleEvent;
 68 
 69             function handleEvent() {
 70                 var returnValue = true;
 71                 // grab the event object (IE uses a global event object)
 72                 var _event = arguments[0] || _this.fixEvent(window.event);
 73                 //注意这里的this指向dom元素 因为addEvent中element["on" + type] = handleEvent
 74                 //获取已经缓存到dom元素的events属性的各个事件函数
 75                 var handlers = this.events[_event.type];
 76                 //遍历已经缓存到元素的事件
 77                 for (var i in handlers) {
 78                     var handler = handlers[i]["handler"];
 79                   
 80                     _event.extendData = handlers[i]["data"];
 81                     this.$$handleEvent = handler;
 82                     if (this.$$handleEvent(_event) === false) {
 83                         returnValue = false;
 84                     }
 85                 }
 86                 return returnValue;
 87             }
 88         },
 89         /**
 90         *反注册事件
 91         *@param {domElement} element dom元素
 92         *@param {String} type 事件类型 
 93         *@param {Function} handler 事件处理函数
 94         *@example
 95         *   SL().Event.removeEvent(tt, "click", see);
 96         */
 97         removeEvent: function (element, type, handler) {
 98             // delete the event handler from the hash table  
 99             if (element.events && element.events[type]) {
100                 delete element.events[type][handler.$$guid];
101             }
102         },
103         fixEvent: function (oEvent) {
104             oEvent.charCode = (oEvent.type == "keypress") ? oEvent.keyCode : 0;
105             oEvent.eventPhase = 2;
106             oEvent.isChar = (oEvent.charCode > 0);
107             oEvent.pageX = oEvent.clientX + document.documentElement.scrollLeft || document.body.scrollLeft;
108             oEvent.pageY = oEvent.clientY + document.documentElement.scrollTop || document.body.scrollTop;
109             oEvent.preventDefault = function () {
110                 this.returnValue = false;
111             };
112 
113             if (oEvent.type == "mouseout") {
114                 oEvent.relatedTarget = oEvent.toElement;
115             } else if (oEvent.type == "mouseover") {
116                 oEvent.relatedTarget = oEvent.fromElement;
117             }
118 
119             oEvent.stopPropagation = function () {
120                 this.cancelBubble = true;
121             };
122 
123             oEvent.target = oEvent.srcElement;
124             oEvent.time = (new Date).getTime();
125             return oEvent;
126         }
127 
128     };
129 
130     SL.Event = SL.Event || {};
131     SL.Event = new Event();
