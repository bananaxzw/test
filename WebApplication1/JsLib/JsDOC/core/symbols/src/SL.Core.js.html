445     function convertToArray(obj, fFilter, oThis) {
446         if (obj == null) return [];
447         if (SL.InstanceOf.Array(obj)) return obj;
448         if (!SL.InstanceOf.Object(obj) && obj != '[object NodeList]') return [obj]; //safari下typeof nodeList 返回 'function'
449         oThis = oThis || window;
450         if (obj.hasOwnProperty) {//非ie浏览器的nodeList集合 都具备 hasOwnProperty方法 
451             if (SL.InstanceOf.DOMNodeList(obj) || SL.InstanceOf.Arguments(obj)) {//ie arguments 可过
452                 return fFilter ? arguments.callee._pushAsArray(obj, fFilter, oThis) : Array.prototype.slice.call(obj);
453             }
454         }
455         else if (SL.InstanceOf.DOMNodeList(obj)) return arguments.callee._pushAsArray(obj, fFilter, oThis); //ie nodeList
456         return arguments.callee._pushAsObject(obj, fFilter, oThis);
457     };
458     convertToArray._pushAsArray = function (obj, fFilter, oThis) {
459         var aReturn = [];
460         if (fFilter) for (var i = 0, len = obj.length; i < len; i++) fFilter.call(oThis, obj[i], i, obj) && aReturn.push(obj[i]);
461         else for (var i = 0, len = obj.length; i < len; i++) aReturn.push(obj[i]);
462         return aReturn;
463     };
464     convertToArray._pushAsObject = function (obj, fFilter, oThis) {
465         var aReturn = [];
466         if (fFilter) for (var o in obj) fFilter.call(oThis, obj[o], o, obj) && aReturn.push(obj[o]);
467         else for (var o in obj) aReturn.push(obj[o]);
468         return aReturn;
469     };

479     /**
480     *@memberOf SL
481     *对象扩展
482     *@example
483     * var object1 = {
484     *        apple: 0,
485     *        banana: { weight: 52, price: 100 },
486     *        cherry: 97
487     *    };
488     *    var object2 = {
489     *        banana: { price: 200 },
490     *        durian: 100
491     *    };
492     *
493     *    console.dir(SL().extend(true, object1, object2));
494     *  {"apple":0,"banana":{"weight":52,"price":200},"cherry":97,"durian":100}
495     *    console.dir(SL().extend(object1, object2));
496     *    {"apple":0,"banana":{"price":200},"cherry":97,"durian":100}
497     */
498     function extend() {
499         var options, name, src, copy, copyIsArray, clone,
500 		target = arguments[0] || {},
501 		i = 1,
502 		length = arguments.length,
503 		deep = false;
504 
505         // Handle a deep copy situation
506         if (typeof target === "boolean") {
507             deep = target;
508             target = arguments[1] || {};
509 
510             i = 2;
511         }
512 
513         // Handle case when target is a string or something (possible in deep copy)
514         if (typeof target !== "object" && (target)) {
515             target = {};
516         }
517 
518         // extend jQuery itself if only one argument is passed
519         if (length === i) {
520             target = SL;
521             --i;
522         }
523 
524         for (; i < length; i++) {
525             if ((options = arguments[i]) != null) {
526                 for (name in options) {
527                     src = target[name];
528                     copy = options[name];
529 
530                     // Prevent never-ending loop
531                     if (target === copy) {
532                         continue;
533                     }
534 
535                     // Recurse if we're merging plain objects or arrays
536                     if (deep && copy && (SL.InstanceOf.PlainObject(copy) || (copyIsArray = SL.InstanceOf.Array(copy)))) {
537                         if (copyIsArray) {
538                             copyIsArray = false;
539                             clone = src && SL.InstanceOf.Array(src) ? src : [];
540 
541                         } else {
542                             clone = src && SL.InstanceOf.PlainObject(src) ? src : {};
543                         }
544 
545                         // Never move original objects, clone them
546                         target[name] = arguments.callee(deep, clone, copy);
547 
548                         // Don't bring in undefined values
549                     } else if (copy !== undefined) {
550                         target[name] = copy;
551                     }
552                 }
553             }
554         }
555 
556         // Return the modified object
557         return target;
558     }
559     SL.extend = extend;
