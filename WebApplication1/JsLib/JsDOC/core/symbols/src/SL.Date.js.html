 49             /// 格式化时间
 50             /// </summary>
 51             /// <param name="date">要格式的日期
 52             /// </param>
 53             /// <param name="format">日期格式 yyyy：4位数的年 yy:年的后2位
 54             /// M：正常表示月份 MM：不足10左边补0 MMM:英文表示 NNN英文缩写

 74             // Convert real date parts into formatted versions

380             /// <param name="date">日期</param>

382             /// <param name="number"></param>

384             if (typeof (part) == "undefined" || part == null || typeof (number) == "undefined" || number == null) {
385                 return date;
386             }
387             number = +number;
388             if (part == 'y') { // year
389                 date.setFullYear(date.getFullYear() + number);
390             }
391             else if (part == 'M') { // Month
392                 date.setMonth(date.getMonth() + number);
393             }
394             else if (part == 'd') { // Day
395                 date.setDate(date.getDate() + number);
396             }
397             else if (part == 'w') { // Weekday工作日 //没测试
398                 var step = (number > 0) ? 1 : -1;
399                 while (number != 0) {
400                     date.add('d', step);
401                     while (date.getDay() == 0 || date.getDay() == 6) {
402                         arguments.callee('d', step);
403                     }
404                     number -= step;
405                 }
406             }
407             else if (part == 'h') { // Hour
408                 date.setHours(date.getHours() + number);
409             }
410             else if (part == 'm') { // Minute
411                 date.setMinutes(date.getMinutes() + number);
412             }
413             else if (part == 's') { // Second
414                 date.setSeconds(date.getSeconds() + number);
415             }
416             return date;
