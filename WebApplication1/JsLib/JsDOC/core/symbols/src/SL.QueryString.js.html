  2 SL().create(function (SL) {
  3 
  4     /**
  5     *查询字符串帮助
  6     *@class
  7     *@name QueryString
  8     *@param {String} str 查询字符串的值
  9     *@example
 10     * window.onload = function () {
 11     *        SL().create(function (ss) {
 12     *            var quering = new ss.QueryString("jj.aspx?name=xuzhiwei&age=11");
 13     *            console.log(quering.name);
 14     *            console.log(quering.age);
 15     *            console.log(quering.toString());
 16     *        });
 17     *    }
 18     */
 19     var QueryString = function (str) {
 20         this.string = str;
 21         this.length = 0;
 22         var re = /[?&]([^=]+)=([^&]*)/g;
 23         var match = re.exec(str);
 24         while (match) {
 25             var name = match[1];
 26             var value = decodeURI(match[2]);
 27             this[name] = value;
 28             this[this.length] = new QueryString.Pair(name, value);
 29             this.length++;
 30             match = re.exec(str);
 31         }
 32     };
 33     /**
 34     *@ignore
 35     */
 36     QueryString.prototype = {
 37         toString: function () {
 38             return this.string;
 39         }
 40     };
 41     /**
 42     *@ignore
 43     */
 44     QueryString.Pair = function (name, value) {
 45         this.name = name;
 46         this.value = value;
 47     };
 48     /**
 49     *@ignore
 50     */
 51     QueryString.Pair.prototype.toString = function () {
 52         return this.name + '=' + this.value;
 53     };
 54     SL.QueryString = QueryString;
 55 });
