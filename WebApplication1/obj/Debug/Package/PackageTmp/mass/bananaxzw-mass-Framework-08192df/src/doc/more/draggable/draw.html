通过拖动画图。 返回

          $.require("ready,more/draggable",function(){
                var canvas
                if(window.VBArray  ){
                    $('pre').html("IE678不支持canvas")
                }else{
                    canvas = $('#draw')[0].getContext('2d');
                    canvas.strokeStyle = "#137";
                    canvas.lineCap = 'round';
                    canvas.lineWidth = 4;
                    canvas.beginPath();
                }
                $('#draw').draggable({
                    lockX: true,
                    lockY: true,
                    dragstart: function( e, dd ){
                        if(canvas){
                            canvas.moveTo( e.pageX - dd.originalX, e.pageY - dd.originalY )
                        }
                    },
                    drag: function( e, dd ){
                        if(canvas){
                            canvas.lineTo( e.pageX - dd.originalX, e.pageY - dd.originalY );
                            canvas.clearRect(0,0,680,300);
                            canvas.stroke();
                        }
                    }
                });
            });
