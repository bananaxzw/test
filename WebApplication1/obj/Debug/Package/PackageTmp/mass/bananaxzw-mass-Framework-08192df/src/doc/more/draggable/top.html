让拖动的元素总是在最上面。 返回

            $.require("ready,more/draggable",function(){
              $(".drag").draggable({//这种方式只适用于拖动的元素都位于同一个父节点,并且z-index都一样的情况
                    dragstart: function(){
                        $( this ).appendTo( this.parentNode );
                    }
                });
            });
