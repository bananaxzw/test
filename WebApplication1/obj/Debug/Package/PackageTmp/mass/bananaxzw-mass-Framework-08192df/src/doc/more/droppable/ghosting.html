拖动左边的方块，在右边的大方块上经过 返回

            $.require("ready,more/droppable",function(){
                $('.drag').droppable({
                    range:".drop",
                    ghosting:true,
                    returning:false,
                   //注意如果为靶场添加太多类名，dropped的样式可能不生效，需要提高优先级，设置important
                    drop: function(event, dd){
                        dd.dropper.toggleClass("dropped");
                    }
                })
            });
