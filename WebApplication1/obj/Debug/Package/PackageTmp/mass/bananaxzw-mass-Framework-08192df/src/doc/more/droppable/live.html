            $.require("ready,more/droppable",function(){
                var num = 1;
                $('#add').click(function(){
                    $('<div class="drop" />')
                    .text( num++ )
                    .appendTo( document.body )
                    .css({
                        top: Math.random() * ( $( window ).height() - 100 ) + 20,
                        left: Math.random() * ( $( window ).width() - 100 ) + 20
                    });
                });

                $('.drag').droppable({
                    range:".drop",
                    activeClass: "active",
                    drop: function(event, dd){
                        dd.dropper.toggleClass("dropped");
                    }
                });
            });
