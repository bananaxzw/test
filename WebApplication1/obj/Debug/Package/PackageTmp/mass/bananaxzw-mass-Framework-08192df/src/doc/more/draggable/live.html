使用事件代理拖动以后才出现的符合要求(指CSS表达式匹配)的元素。 返回

            $.require("ready,more/draggable",function(){
                var num = 1;
                $('#add').click(function(){
                    $('<div class="drag"/>')
                    .text( num++ )
                    .appendTo( document.body )
                    .css({
                        top: Math.random() * ( $( window ).height() - 100 ) + 20,
                        left: Math.random() * ( $( window ).width() - 100 ) + 20
                    });
                });
                $('.drag').draggable({live:true})
            });
