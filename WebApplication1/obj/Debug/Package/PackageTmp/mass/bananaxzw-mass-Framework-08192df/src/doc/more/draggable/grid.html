一个个格子地移动。 返回

            $.require("ready,more/draggable",function(){
                $(".drag").draggable({
                    lockX: true, //阻止系统默认的移动方式
                    lockY: true, //阻止系统默认的移动方式
                    drag:function(e, dd){//使用drag回调，自己设置如何移动
                        $( this ).css({
                            top: Math.round( dd.offsetY / 20 ) * 20,
                            left: Math.round( dd.offsetX / 20 ) * 20
                        });
                    }
                });
            });
