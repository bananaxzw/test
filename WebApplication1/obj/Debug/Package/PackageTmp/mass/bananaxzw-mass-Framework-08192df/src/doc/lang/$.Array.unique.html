array.unique()

描述：

对数组进行去重,但不影响原对象。

各种实现：

            //方法1
            Array.prototype.unique = function () {
                var ret = [];
                    loop:for(var i = 0, n = this.length; i < n; i++) {
                        for(var x = i + 1 ; x < n; x++) {
                            if(this[x] === this[i])
                                continue loop;
                        }
                        ret.push(this[i]);
                    }
                return ret;
            }
            //方法2
            Array.prototype.unique = function () {
                var ret = [],ri = 0, array = this.concat().sort();
                ret[ri] = array[0];
                for(var j = 1, n = array.length; j < n; j++){
                    if(ret[ri] !== array[j]){
                        ret[++ri] = array[j]
                    }
                }
                return ret;
            }
